# Finding limit of function
from sympy import sin, cos, limit
f = sin(x)/x
limit(f,x,0)
Out[3]:
$$1$$
In [4]:
from sympy import diff
f = sin(x)/x
g = diff(f, x)
g
Out[4]:
$$\frac{\cos{\left (x \right )}}{x} - \frac{\sin{\left (x \right )}}{x^{2}}$$
In [5]:
% matplotlib inline

from sympy.plotting import plot

f = sin(x)/x
g = diff(f, x)

p1 = plot(f, show=False)
p2 = plot(g, line_color='red', show=False)
p1.append(p2[0])
p1.show()

Differentiating using Definition¶

In [6]:
x, h = symbols('x h')
f = x**n
Δf = (x+h)**n - x**n
limit(Δf/h, h, 0)
Out[6]:
$$\frac{n x^{n}}{x}$$
In [7]:
f = lambda x: sin(x) * cos(x)
def Δ(f):
    return f(x+h) - f(x)
limit(Δ(f)/h, h, 0)
Out[7]:
$$- \sin^{2}{\left (x \right )} + \cos^{2}{\left (x \right )}$$
